# install node exporter on Linux machine for prmetheus datasource

1, get the latest binary file from the official repo,  https://github.com/prometheus/node_exporter/releases

    # wget https://github.com/prometheus/node_exporter/releases/download/v1.9.1/node_exporter-1.9.1.linux-arm64.tar.gz

    # tar xvf node_exporter-1.9.1.linux-arm64.tar.gz

    # mv node_exporter /usr/local/bin 

2, configure node exporter as a service

    # create service user and service file
        * useradd -rs /bin/false node_exporter
        * sudo vi /etc/systemd/system/node_exporter.service

    # copy the following systemd config file 

        [Unit]
        Description=Node Exporter
        After=network.target

        [Service]
        User=node_exporter
        Group=node_exporter
        Type=simple
        ExecStart=/usr/local/bin/node_exporter

        [Install]
        WantedBy=multi-user.target
    
    # enable and start the service

        * systemctl enable --now node_exporter.service
